-- draggable
do
	local dragging = false
	local offset
	panel.InputBegan:Connect(function(input)
		if input.UserInputType.Name == "MouseButton1" or input.UserInputType.Name == "Touch" then
			dragging = true
			offset = input.Position
		end
	end)
	panel.InputEnded:Connect(function(input)
		if input.UserInputType.Name == "MouseButton1" or input.UserInputType.Name == "Touch" then
			dragging = false
		end
	end)
	game:GetService("UserInputService").InputChanged:Connect(function(input)
		if dragging and (input.UserInputType.Name == "MouseMovement" or input.UserInputType.Name == "Touch") then
			local delta = input.Position - offset
			offset = input.Position
			panel.Position = UDim2.new(panel.Position.X.Scale, panel.Position.X.Offset + delta.X,
				panel.Position.Y.Scale, panel.Position.Y.Offset + delta.Y)
		end
	end)
end

-- fake pet “shadow” near the player (pure client visual)
local function showFakeShadow(name)
	local char = player.Character or player.CharacterAdded:Wait()
	if not char:FindFirstChild("HumanoidRootPart") then return end
	local attach = Instance.new("Part")
	attach.Name = "PetShadow"
	attach.Size = Vector3.new(1,1,1)
	attach.Transparency = 1
	attach.Anchored = false
	attach.CanCollide = false
	attach.Parent = workspace
	local weld = Instance.new("WeldConstraint", attach)
	weld.Part0 = attach
	weld.Part1 = char:WaitForChild("HumanoidRootPart")
	attach.CFrame = char.HumanoidRootPart.CFrame * CFrame.new(2, 0, -2)

	local billboard = Instance.new("BillboardGui", attach)
	billboard.Size = UDim2.fromOffset(140, 40)
	billboard.StudsOffset = Vector3.new(0, 3, 0)
	billboard.AlwaysOnTop = true

	local tag = Instance.new("TextLabel", billboard)
	tag.Size = UDim2.fromScale(1,1)
	tag.BackgroundTransparency = 1
	tag.Font = Enum.Font.GothamBold
	tag.TextSize = 16
	tag.TextColor3 = Color3.fromRGB(255, 215, 120)
	tag.TextStrokeTransparency = 0.5
	tag.Text = name .. "  (client)"
end

-- fake console log helper
local function log(lineText)
	console.Text = (console.Text .. "\n" .. lineText):sub(-1200)
end

-- progress anim
local function animateProgress(t)
	fill.Size = UDim2.new(0, 0, 1, 0)
	local tw = TweenService:Create(fill, TweenInfo.new(t, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(1, 0, 1, 0)})
	tw:Play()
end

-- prank reveal
local revealed = false
local function reveal()
	if revealed then return end
	revealed = true
	notify("pranked", "this is a pet spawner visual!")
	log("[WARN] server rejected instance build id mismatch")
	log("[INFO] visual mode detected applying client overlay only")
	title.Text = "Adrian Spawner • VISUAL ONLY"
end

-- main fake flow
local function fakeSpawn(mult)
	mult = mult or 1
	local petName = petInput.Text ~= "" and petInput.Text or "Kitsune"
	local kg = tonumber(kgInput.Text) or math.random(3, 150)
	local age = tonumber(ageInput.Text) or math.random(1, 50)

	log(string.format("[INFO] preparing blueprint  pet=%s  kg=%d  age=%d", petName, kg, age))
	log("[SEC] validating client token...")
	animateProgress(1.2); task.wait(0.6); log("[OK] token accepted")
	task.wait(0.7)
	log("[NET] requesting server pipeline..."); task.wait(0.5)
	log("[OK] pipeline route=egg/spawn v2"); task.wait(0.4)
	log("[BUILD] composing mesh & traits..."); task.wait(0.8)
	log("[SYNC] pushing " .. tostring(mult) .. " instance(s)..."); task.wait(0.6)
	log("[SYNC] awaiting ack..."); task.wait(0.7)

	-- look super legit
	notify("spawning", petName .. " x" .. tostring(mult) .. " • kg " .. kg .. " • age " .. age, 3)
	showFakeShadow(petName)

	-- after a moment show “success” then quietly reveal prank
	task.wait(0.8)
	log("[OK] ack received id=" .. string.format("%06d", math.random(100000, 999999)))
	task.wait(0.4)
	log("[DONE] spawn complete")
	notify("success", petName .. " spawned locally", 3)

	-- reveal after 20s automatically so the joke lands later
	task.delay(20, reveal)
end
